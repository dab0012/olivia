 99    def initialize(self, package_names: list[str] = None, show_progress: bool = False, chunk_size: int = 10000):
100        '''
101        Initializes the package manager by loading the packages from the data source
102
103        Parameters
104        ----------
105        package_list : list[str]
106            List of package names to load, if None, all the packages will be loaded
107        show_progress : bool
108            If True, a progress bar will be shown
109        chunk_size : int
110            Size of the chunks to load the packages, this is done to avoid memory errors            
111        '''
112
113        # Get package names from the data sources if needed
114        if package_names is None:
115            for data_source in self.data_sources:
116                try:
117                    package_names = data_source.obtain_package_names()
118                    break
119                except NotImplementedError:
120                    MyLogger().get_logger().debug(f"Data source {data_source.name} does not implement obtain_package_names method")
121                    continue
122                except Exception as e:
123                    MyLogger().get_logger().error(f"Error while obtaining package names from data source {data_source.name}: {e}")
124                    continue
125
126        # Check if the package names are valid
127        if package_names is None or not isinstance(package_names, list):
128            raise ValueError("No valid package names found")
129        
130        # Instantiate the progress bar if needed
131        progress_bar = tqdm.tqdm(
132            total=len(package_names),
133            colour="green",
134            desc="Loading packages",
135            unit="packages",
136        ) if show_progress else None
137
138        # Create a chunked list of package names
139        # This is done to avoid memory errors
140        package_names_chunked = [package_names[i:i + chunk_size] for i in range(0, len(package_names), chunk_size)]
141
142        for package_names in package_names_chunked:
143            # Obtain the packages data from the data source and store them
144            self.obtain_packages(
145                package_names=package_names, 
146                progress_bar=progress_bar,
147                extend=True
148            )
149        
150        # Close the progress bar if needed
151        if progress_bar is not None:
152            progress_bar.close()

155    def obtain_package(self, package_name: str) -> Union[Package, None]:
156        '''
157        Builds a Package object using the data sources in order until one of them returns a valid package
158
159        Parameters
160        ----------
161        package_name : str
162            Name of the package
163
164        Returns
165        -------
166        Union[Package, None]
167            Package object if the package exists, None otherwise
168
169        Examples
170        --------
171        >>> package = package_manager.obtain_package("package_name")
172        >>> package
173        <Package: package_name>
174        '''
175        # Obtain the package data from the data sources in order
176        package_data = None
177        for data_source in self.data_sources:
178            package_data = data_source.obtain_package_data(package_name)
179            if package_data is not None:
180                MyLogger().get_logger().debug(f"Package {package_name} found using {data_source.name}")
181                break
182            else:
183                MyLogger().get_logger().debug(f"Package {package_name} not found using {data_source.name}")
184
185        # Return the package if it exists
186        return None if package_data is None else Package.load(package_data)

188    def obtain_packages(
189        self,
190        package_names: Optional[list[str]],
191        progress_bar: Optional[tqdm.tqdm],
192        extend: bool = False
193    ) -> list[Package]:
194        '''
195        Builds a list of Package objects using the data sources in order until one of them returns a valid package
196
197        Parameters
198        ----------
199        package_names : Optional[list[str]]
200            List of package names to load, if None, all the packages will be loaded
201        progress_bar : Optional[tqdm.tqdm]
202            Progress bar to show the progress of the loading
203        extend : bool
204            If True, the packages will be added to the package manager
205
206        Returns
207        -------
208        list[Package]
209            List of Package objects
210            
211        Examples
212        --------
213        >>> packages = package_manager.obtain_packages(["package_name_1", "package_name_2"])
214        >>> packages
215        [<Package: package_name_1>, <Package: package_name_2>]
216        '''
217        
218        # Check if the package names are valid
219        if package_names is not None and not isinstance(package_names, list):
220            raise ValueError("Package names must be a list")
221
222        # Obtain the packages data from the data source
223        pending_packages = package_names.copy()
224        preferred_data_source = self.data_sources[0]
225
226        # Return list
227        packages = []
228
229        # if datasource is instance of ScraperDataSource use the obtain_packages_data method for parallelization
230        if isinstance(preferred_data_source, ScraperDataSource):
231            packages_data = []
232            data_found, not_found = preferred_data_source.obtain_packages_data(
233                package_names=pending_packages, 
234                progress_bar=progress_bar
235            )
236            packages_data.extend(data_found)
237            # pending_packages = not_found
238            MyLogger().get_logger().info(f"Packages found: {len(data_found)}")
239            MyLogger().get_logger().info(f"Packages not found: {len(not_found)}")
240            packages = [Package.load(package_data) for package_data in packages_data]
241
242        # if not use the obtain_package_data method for sequential processing using the data_sources of the list
243        else:
244
245            while len(pending_packages) > 0:
246                
247                package_name = pending_packages[0]
248                package_data = self.obtain_package(package_name)
249                if package_data is not None:
250                    packages.append(package_data)
251
252                # Remove the package from the pending packages
253                del pending_packages[0]
254
255                if progress_bar is not None:
256                    progress_bar.update(1)
257        
258        MyLogger().get_logger().info(f"Total packages found: {len(packages)}")
259        
260
261        # update the self.packages attribute overwriting the packages with the same name
262        # but conserving the other packages
263        if extend:
264            MyLogger().get_logger().info("Extending data source with obtained packages")
265            for package in packages:
266                self.packages[package.name] = package
267
268        return packages

282        Load a csv file into a PackageManager object
283
284        Parameters
285        ----------
286        csv_path : str
287            Path of the csv file to load
288        dependent_field : str = None, optional
289            Name of the dependent field, by default None
290        dependency_field : str = None, optional
291            Name of the dependency field, by default None
292        version_field : str = None, optional
293            Name of the version field, by default None
294        dependency_version_field : str = None, optional
295            Name of the dependency version field, by default None

298        default_format : bool, optional

306            dependency_field="dependency",
307            version_field="version",
308            dependency_version_field="dependency_version",
309            url_field="url"
310        )
311        >>> pm = PackageManager.load_csv_adjlist("full_adjlist.csv", default_format=True)
312
313        '''
314        try:
315            data = pd.read_csv(csv_path)
316        except Exception as e:
317            raise PackageManagerLoadError(
318                f"Error loading csv file: {e}") from e
319
320        csv_fields = []
321
322        if default_format:
323            # If the csv has the structure of full_adjlist.csv, we use the default fields
324            dependent_field = 'name'
325            dependency_field = 'dependency'
326            version_field = 'version'
327            dependency_version_field = 'dependency_version'
328            url_field = 'url'
329            csv_fields = [dependent_field, dependency_field,
330                        version_field, dependency_version_field, url_field]
331        else:
332            if dependent_field is None or dependency_field is None:
333                raise PackageManagerLoadError(
334                    "Dependent and dependency fields must be specified")
335
336            csv_fields = [dependent_field, dependency_field]
337            # If the optional fields are specified, we add them to the list
338            if version_field is not None:
339                csv_fields.append(version_field)
340            if dependency_version_field is not None:
341                csv_fields.append(dependency_version_field)
342            if url_field is not None:
343                csv_fields.append(url_field)
344
345        # If the csv does not have the specified fields, we raise an error
346        if any(col not in data.columns for col in csv_fields):
347            raise PackageManagerLoadError("Invalid csv format")
348
349        # We create the data source
350        data_source = CSVDataSource(
351            file_path=csv_path,
352            name="Package Manager CSV",
353            description="Package Manager obtained from a csv file",
354            dependent_field=dependent_field,
355            dependency_field=dependency_field,
356            dependent_version_field=version_field,
357            dependency_version_field=dependency_version_field,
358            dependent_url_field=url_field
359        )
360
361        # We create the package manager
362        return cls(data_source)

415    def export_adjlist(self, full_data = False) -> pd.DataFrame:
416        '''
417        Convert the object to a adjacency list, where each row represents a dependency
418        If a package has'nt dependencies, it will appear in the list with dependency field empty
419
420        Parameters
421        ----------
422        full_data : bool, optional
423            If True, the adjacency list will contain the version and url of the packages, by default False
424
425        Returns
426        -------
427        pd.DataFrame
428            Dependency network as an adjacency list
429
430        Examples
431        --------
432        >>> adj_list = package_manager.export_adjlist()
433        >>> print(adj_list)
434            [name, dependency]
435        '''
436
437        if not self.packages:
438            MyLogger().get_logger().debug("The package manager is empty")
439            return pd.DataFrame()
440                    
441
442        rows = []
443
444        if full_data:
445            for package_name in self.packages.keys():
446                package = self.get_package(package_name)
447
448
449                for dependency in package.dependencies:
450                    
451                    try:
452                        dependency_full = self.get_package(dependency.name)
453                        rows.append(
454                            [package.name, package.version, package.url, dependency_full.name, dependency_full.version, dependency_full.url]
455                        )
456                    except Exception:
457                        if dependency.name is not None:
458                            rows.append(
459                                [package.name, package.version, package.url, dependency.name, None, None]
460                            )
461
462
463            return pd.DataFrame(rows, columns=['name', 'version', 'url', 'dependency', 'dependency_version', 'dependency_url'])
464        else:
465            for package_name in self.packages.keys():
466                package = self.get_package(package_name)
467                rows.extend(
468                    [package.name, dependency.name]
469                    for dependency in package.dependencies
470                )
471            return pd.DataFrame(rows, columns=['name', 'dependency'])

473    def dependency_network(
474            self, 
475            package_name: str, 
476            dependency_network: dict = None, 
477            deep_level: int = 5,
478            generate = False
479        ) -> dict[str, list[Package]]:
480        """
481        Generates the dependency network of a package from the data source.
482
483        Parameters
484        ----------
485        package_name : str
486            The name of the package to generate the dependency network
487        dependency_network : dict, optional
488            The dependency network of the package            
489        deep_level : int, optional
490            The deep level of the dependency network, by default 5
491        generate : bool, optional
492            If True, the dependency network will be generated from the data source, by default False
493
494        Returns
495        -------
496        dict[str, list[Package]]
497            The dependency network of the package
498        """
499
500        # If the deep level is 0, we return the dependency network (Stop condition)
501        if deep_level == 0:
502            return dependency_network
503        
504        # If the dependency network is not specified, we create it (Initial case)
505        if dependency_network is None:
506            dependency_network = {}
507
508        # If the dependency network is not specified, we create it obtaining the data from the package manager
509        if generate:
510            return self._generate_dependency_network(package_name, deep_level, dependency_network)
511        
512        # Use the data of the package manager
513        current_package = self.get_package(package_name)
514        dependencies = []
515        if (current_package is not None) and (package_name not in dependency_network):
516
517            # Get the dependencies of the package and add it to the dependency network if it is not already in it
518            dependencies = current_package.get_dependencies_names()
519            if dependencies:
520                dependency_network[package_name] = dependencies
521            
522        # Append the dependencies of the package to the dependency network
523        for dependency_name in dependencies:
524            if (dependency_name not in dependency_network) and  (self.get_package(dependency_name) is not None):
525                    
526                dependency_network = self.dependency_network(
527                    package_name = dependency_name, 
528                    dependency_network = dependency_network, 
529                    deep_level = deep_level - 1,
530                    generate = False
531                )
532    
533        return dependency_network

535    def _generate_dependency_network(self, package_name: str, deep_level: int = 5, dependency_network: dict = None) -> dict[str, list[Package]]:
536        """
537        Generates the dependency network of a package from the data source.
538
539        Parameters
540        ----------
541        package_name : str
542            The name of the package to generate the dependency network
543        deep_level : int, optional
544            The deep level of the dependency network, by default 5
545        dependency_network : dict, optional
546            The dependency network of the package
547
548        Returns
549        -------
550        dict[str, list[Package]]
551            The dependency network of the package
552        """
553        
554        # Obtain the data of target package
555        try:
556            current_package = self.obtain_package(package_name)
557        except ValueError:
558            MyLogger().get_logger().debug(
559                f"The package {package_name} does not exist in the data source {self.name}"
560            )
561            return dependency_network
562
563        # Get the dependencies of the package and add it to the dependency network if it is not already in it
564        dependencies = current_package.get_dependencies_names()
565
566        if (package_name not in dependency_network) and dependencies:
567            dependency_network[package_name] = []
568
569            # Append the dependencies of the target package to the dependency network
570            for dependency_name in dependencies:
571                dependency_network[package_name].append(dependency_name)
572
573                if dependency_name in dependency_network:
574                    continue
575                try:     
576                    self.dependency_network(
577                        dependency_name,                # The name of the dependency
578                        dependency_network,             # The global dependency network
579                        deep_level - 1,                 # The deep level is reduced by 1
580                        generate = True                 # The dependency network is generated
581                    )
582
583                except Exception:
584                    MyLogger().get_logger().debug(
585                        f"The package {dependency_name}, as dependency of {package_name} does not exist in the data source {self.name}"
586                    )
587        return dependency_network

474    def dependency_network(
475            self, 
476            package_name: str, 
477            dependency_network: dict = None, 
478            deep_level: int = 5,
479            generate = False
480        ) -> dict[str, list[Package]]:
481        """
482        Generates the dependency network of a package from the data source.
483
484        Parameters
485        ----------
486        package_name : str
487            The name of the package to generate the dependency network
488        dependency_network : dict, optional
489            The dependency network of the package            
490        deep_level : int, optional
491            The deep level of the dependency network, by default 5
492        generate : bool, optional
493            If True, the dependency network will be generated from the data source, by default False
494
495        Returns
496        -------
497        dict[str, list[Package]]
498            The dependency network of the package
499        """
500
501        # If the deep level is 0, we return the dependency network (Stop condition)
502        if deep_level == 0:
503            return dependency_network
504        
505        # If the dependency network is not specified, we create it (Initial case)
506        if dependency_network is None:
507            dependency_network = {}
508
509        # If the dependency network is not specified, we create it obtaining the data from the package manager
510        if generate:
511            return self._generate_dependency_network(package_name, deep_level, dependency_network)
512        
513        # Use the data of the package manager
514        current_package = self.get_package(package_name)
515        dependencies = []
516        if (current_package is not None) and (package_name not in dependency_network):
517
518            # Get the dependencies of the package and add it to the dependency network if it is not already in it
519            dependencies = current_package.get_dependencies_names()
520            if dependencies:
521                dependency_network[package_name] = dependencies
522            
523        # Append the dependencies of the package to the dependency network
524        for dependency_name in dependencies:
525            if (dependency_name not in dependency_network) and  (self.get_package(dependency_name) is not None):
526                    
527                dependency_network = self.dependency_network(
528                    package_name = dependency_name, 
529                    dependency_network = dependency_network, 
530                    deep_level = deep_level - 1,
531                    generate = False
532                )
533    
534        return dependency_network

536    def _generate_dependency_network(self, package_name: str, deep_level: int = 5, dependency_network: dict = None) -> dict[str, list[Package]]:
537        """
538        Generates the dependency network of a package from the data source.
539
540        Parameters
541        ----------
542        package_name : str
543            The name of the package to generate the dependency network
544        deep_level : int, optional
545            The deep level of the dependency network, by default 5
546        dependency_network : dict, optional
547            The dependency network of the package
548
549        Returns
550        -------
551        dict[str, list[Package]]
552            The dependency network of the package
553        """
554        
555        # Obtain the data of target package
556        try:
557            current_package = self.obtain_package(package_name)
558        except ValueError:
559            MyLogger().get_logger().debug(
560                f"The package {package_name} does not exist in the data source {self.name}"
561            )
562            return dependency_network
563
564        # Get the dependencies of the package and add it to the dependency network if it is not already in it
565        dependencies = current_package.get_dependencies_names()
566
567        if (package_name not in dependency_network) and dependencies:
568            dependency_network[package_name] = []
569
570            # Append the dependencies of the target package to the dependency network
571            for dependency_name in dependencies:
572                dependency_network[package_name].append(dependency_name)
573
574                if dependency_name in dependency_network:
575                    continue
576                try:     
577                    self.dependency_network(
578                        dependency_name,                # The name of the dependency
579                        dependency_network,             # The global dependency network
580                        deep_level - 1,                 # The deep level is reduced by 1
581                        generate = True                 # The dependency network is generated
582                    )
583
584                except Exception:
585                    MyLogger().get_logger().debug(
586                        f"The package {dependency_name}, as dependency of {package_name} does not exist in the data source {self.name}"
587                    )
588        return dependency_network
